def MarketCapConvert(old_format):
    '''
    Converts a monetary value format into float
    Ex: $1.17B --> 1170000000
    '''
    M_cap = str(old_format)
    if M_cap == "n/a":
        return 0
    else:
        Million=True
        if 'B' in M_cap:
            Million=False
        M_cap = M_cap.replace('$','')
        M_cap = M_cap.replace('M','')
        M_cap = M_cap.replace('B','')
        M_cap = float(M_cap)
        if Million:
            M_cap *= 1000000
        else:
            M_cap *= 1000000000
        return M_cap
